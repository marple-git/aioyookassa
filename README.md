    print(f"Confirmation URL: {payment.confirmation.url}")

    # –ü–æ–ª—É—á–µ–Ω–∏–µ —Å–ø–∏—Å–∫–∞ –ø–ª–∞—Ç–µ–∂–µ–π –∑–∞ —Å–µ–≥–æ–¥–Ω—è (–∏—Å–ø–æ–ª—å–∑—É–µ–º Pydantic –º–æ–¥–µ–ª—å)
